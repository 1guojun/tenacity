#include<stdio.h>
void MaoPao(int arr[], int len)
{
    int j,i;
    for ( i = 0; i < len - 1; i++)
    {
        int ischange = 0;
        for ( j = 0; j < len -1- i; j++)
        {
            if (arr[j] < arr[j + 1])
            {
                int tmp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = tmp;
                ischange = 1;//进行优化如果 如果本次冒泡中，元素没有交换，则本次开始冒泡时，数据已经有序了，后面的冒泡可以不用进行了
            }
        }
        if (!ischange)
            return;
    }
    for (j = 0; j < len ; j++)
    {
        printf("%d ", arr[j]);
    }
}
int main()
{
    int arr[] = { 3, 2, 5,8,4, 7, 6, 9 };
    int len = sizeof(arr) / sizeof(arr[0]);
    MaoPao(arr, len);
    return 0;
}
